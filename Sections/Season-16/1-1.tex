\SSpsetQ{

If $n^2\%$ of $n$ is 80, find $n$.
}\bigskip

\begin{solution}\hfil\medskip
  
    We have $$\frac{n^2}{100} \cdot n = 80 \iff n = \boxed{20}.$$
\end{solution}\bigskip
