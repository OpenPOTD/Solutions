\SSpsetQ{
	For an integer $n$, let $d(n)$ be the sum of integer divisors of $n$. Find the sum of all 
	$1 < n < 100$ such that $n$ maximizes $d(n)$. 
	
	%Put Problem Here
}\bigskip

\begin{solution}\hfil\medskip
	
	If $d|n$ then $-d|n$, so $d(n) = 0$ for all $n$. Our answer is \fbox{4949}.
	%Put sol here
\end{solution}\bigskip
